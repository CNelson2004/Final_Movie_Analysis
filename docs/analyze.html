from scipy import stats
import pingouin as pg
df = pd.read_csv("movie_data.csv")
print("Size of dataset before dropping NA values for Season and Total Box Office Revenue")
print(len(df))
df = df.dropna(subset=['Season'])
df = df.dropna(subset=['Total Box Office Revenue'])
print("Size of dataset after dropping NA values")
print(len(df))
print("Size of groups within Season")
print(df["Season"].value_counts())
groups = [df[df['Season'] == s]['Total Box Office Revenue']
        for s in df['Season'].unique()]
f_stat, p_value = stats.f_oneway(*groups)
print("F value and p value of Season as in relation to Total Box Office Revenue")
print("F=",f_stat,"p =", p_value)
gh = pg.pairwise_gameshowell(dv='Total Box Office Revenue',between="Season",data=df)
print("Actual significance between groups")
print(gh)
